
namespace = mafoe_gain_dev

# Gain tax
# positive factors: population growth and government/tax installations
# 	- terrain allowing high population: grasslands, farmlands
#   - being the capital
#   - has government or taxation buildings (the more, the better)
#
# negative factors: anything negative to population growth or stability
#   - terrain: mountain, desert, marsh, highlands, jungle, coastal_desert
#   - climate: tropical, severe_winter, arctic
#   - is looted
#   - is a colony
#   - unrest (the higher, the worse)
#   - different religion
#   - different culture

province_event = {
	id = mafoe_gain_dev.1
	title = "mafoe.gain_tax.title"
	desc = "mafoe.gain_tax.desc"
	picture = CITY_DEVELOPMENT_eventPicture

	hidden = no

	trigger = {
            is_empty = no
            is_sea = no
            is_wasteland = no
	}

	mean_time_to_happen = {
		months = 12

        # faster in the capital
		modifier = {
			factor = 0.5
            is_capital = yes
		}

        # faster in favourable terrain
		modifier = {
			factor = 0.9
			OR = {
				has_terrain = grasslands
				has_terrain = farmlands
			}
		}

        # slower in unsuitable terrain
		modifier = {
			factor = 1.15
			OR = {
				has_terrain = jungle
				has_terrain = marsh
			}
		}

        # slower in hostile terrain
		modifier = {
			factor = 1.5
			OR = {
				has_terrain = glacier
				has_terrain = mountain
				has_terrain = desert
				has_terrain = coastal_desert
			}
		}

        # slower in unsuitable climate
		modifier = {
			factor = 1.5
			OR = {
				has_climate = arid
				has_climate = tropical
			}
		}

        # slower in hostile terrain
		modifier = {
			factor = 2
			has_climate = arctic
		}
        # slower with more unrest
		modifier = {
			factor = 2
			unrest = 2
		}
		modifier = {
			factor = 2
			unrest = 4
		}
		modifier = {
			factor = 2
			unrest = 6
		}
		modifier = {
			factor = 2
			unrest = 8
		}
		modifier = {
			factor = 2
			unrest = 10
		}

        # faster with tax or government buildings
		modifier = {
			factor = 0.9
			has_building = temple
		}
		modifier = {
			factor = 0.9
			has_building = cathedral
		}
		modifier = {
			factor = 0.9
			has_building = courthouse
		}
		modifier = {
			factor = 0.9
			has_building = town_hall
		}

		# slower when looted or blockaded
		modifier = {
			factor = 5
            OR = {
                is_looted = yes
                is_blockaded = yes
            }
		}

		# slower when colony
		modifier = {
			factor = 5
            is_colony = yes
		}

		# faster when tradegood is wheat, fish, salt or slaves
		modifier = {
			factor = 5
			OR = {
                trade_goods = grain
                trade_goods = fish
                trade_goods = salt
            }
		}

		# faster when taxation boost is active
		modifier = {
			factor = 5
			owner = {
    			has_country_modifier = tax_growth_boosted
			}
		}

		# faster with higher development efficiency
		# has 3 tiers from adm tech, each is 0.1 or 10%
		modifier = {
			factor = 2
			is_variable_equal = { which = development_efficiency value = 0.3 }
		}
		modifier = {
			factor = 2
			is_variable_equal = { which = development_efficiency value = 0.2 }
		}
		modifier = {
			factor = 2
			is_variable_equal = { which = development_efficiency value = 0.1 }
		}

        # faster with higher local development cost
        # -20% from university
        # -5% through from aspect
        # -10% from custom idea
        # +50% from tribe disaster
        # +10% or -10% from burgher estate
        # -X% from harbor, center of trade, etc
        # -X% from events
        # -X% from governments
        # -20% from economic ideas
        # -5-10% from national ideas
        # -X% from institutions, policies, parliament issues, buddhism, terrain, difficulty, being subject or primitive, trade goods,
        modifier = {
            factor = 5
            has_local_modifier_value = {
                which = local_development_cost
                value = 0.15
            }
        }
        modifier = {
            factor = 5
            has_global_modifier_value = {
                which = development_cost
                value = 0.15
            }
        }
	}

	option = {
		name = "mafoe.gain_tax.option"
		add_base_tax = 377
		add_base_manpower = 377
		add_base_production = 377
	}
}
